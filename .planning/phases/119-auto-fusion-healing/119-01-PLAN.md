```python
def is_parameter_locked(locks: dict[str, float], parameter: str) -> bool:
    """Check if a parameter is in hysteresis cooldown.
    Args:
        locks: Dict mapping parameter name to monotonic expiry time.
        parameter: Parameter name to check.
    Returns:
        True if locked (cooldown active), False if unlocked or expired.
    """
    expiry = locks.get(parameter)
    if expiry is None:
        return False
    if time.monotonic() >= expiry:
        del locks[parameter]
        return False
    return True

def lock_parameter(locks: dict[str, float], parameter: str, cooldown_sec: float) -> None:
    """Lock a parameter for a cooldown period after revert.
    Args:
        locks: Dict mapping parameter name to monotonic expiry time.
        parameter: Parameter name to lock.
        cooldown_sec: Cooldown duration in seconds.
    """
    locks[parameter] = time.monotonic() + cooldown_sec
```

    **FusionHealer class (per D-01 through D-07):**

    Constructor `__init__(self, wan_name: str, *, suspend_threshold: float = 0.3, recover_threshold: float = 0.5, suspend_window_sec: float = 60.0, recover_window_sec: float = 300.0, grace_period_sec: float = 1800.0, cycle_interval_sec: float = 0.05, min_samples: int = 100, alert_engine: AlertEngine | None = None, parameter_locks: dict[str, float] | None = None) -> None`:
    - Store all config params
    - `self._state = HealState.ACTIVE`
    - `self._suspend_window_samples = int(suspend_window_sec / cycle_interval_sec)` (1200 at 20Hz)
    - `self._recover_window_samples = int(recover_window_sec / cycle_interval_sec)` (6000 at 20Hz)
    - `self._grace_until: float = 0.0` (monotonic expiry)
    - `self._min_samples = min_samples`
    - Rolling Pearson buffers: `self._x_buf: deque[float] = deque(maxlen=self._suspend_window_samples)`, same for `_y_buf`
    - Running sums: `_sum_x`, `_sum_y`, `_sum_xy`, `_sum_x2`, `_sum_y2` all float = 0.0, `_n = 0`
    - Sustained counters: `_sustained_below_count = 0`, `_sustained_above_count = 0`
    - `_last_transition_ts: float | None = None`
    - `_alert_engine = alert_engine`
    - `_parameter_locks = parameter_locks` (reference to WANController._parameter_locks dict)

    **`_add_sample(self, x: float, y: float) -> None`:**
    - If deque full, evict oldest: subtract old_x/old_y from running sums, decrement _n
    - Append new x/y to deques, add to running sums, increment _n

    **`_compute_pearson(self) -> float | None`:**
    - Return None if `self._n < self._min_samples`
    - Use incremental formula: `numerator = n*sum_xy - sum_x*sum_y`, `denom_sq = (n*sum_x2 - sum_x**2) * (n*sum_y2 - sum_y**2)`
    - If `denom_sq <= 0.0` return 0.0 (constant signal)
    - Return `numerator / math.sqrt(denom_sq)`

    **`tick(self, icmp_delta: float, irtt_delta: float) -> HealState`:**
    - Call `_add_sample(icmp_delta, irtt_delta)`
    - If `time.monotonic() < self._grace_until`: return current state (grace active, skip monitoring)
    - `r = self._compute_pearson()`
    - If r is None: return current state (warmup)
    - State machine logic:
      - **ACTIVE:** if `r < self._suspend_threshold`: increment `_sustained_below_count`; if count >= `_suspend_window_samples`: transition to SUSPENDED, fire alert, lock param, reset counters. Else: reset `_sustained_below_count` to 0.
      - **SUSPENDED:** if `r >= self._recover_threshold`: increment `_sustained_above_count`; if count >= `_recover_window_samples`: transition to RECOVERING, fire alert, reset counters. Else: reset `_sustained_above_count` to 0.
      - **RECOVERING:** if `r >= self._recover_threshold`: increment `_sustained_above_count`; if count >= `_recover_window_samples`: transition to ACTIVE, fire alert, unlock param, reset counters. Else: reset `_sustained_above_count` to 0; if `r < self._suspend_threshold`: transition back to SUSPENDED (correlation worsened during recovery), fire alert, reset counters.
    - Return `self._state`

    Note on RECOVERING: Per the research Pitfall 5, the parameter lock must persist through RECOVERING. The lock is set on ACTIVE->SUSPENDED and only cleared on transition to ACTIVE. RECOVERING->SUSPENDED does not need to re-lock (already locked). The `_sustained_above_count` counter in RECOVERING accumulates toward the `_recover_window_samples` threshold.

    **`_fire_transition_alert(self, alert_type: str, r: float) -> None`:**
    - If `self._alert_engine` is None, return
    - Call `self._alert_engine.fire(alert_type=alert_type, severity="warning", wan_name=self._wan_name, details={"pearson_r": round(r, 4), "threshold": self._suspend_threshold if "suspend" in alert_type else self._recover_threshold, "state": self._state.value}, rule_key="fusion_healing")`

    **`start_grace_period(self) -> None` (per D-06):**
    - `self._grace_until = time.monotonic() + self._grace_period_sec`
    - `self._sustained_below_count = 0`
    - `self._sustained_above_count = 0`

    - `self._state = HealState.ACTIVE`
    - `self._last_transition_ts = time.monotonic()`
