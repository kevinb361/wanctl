```python
def measure_rtt(self) -> float | None:
    """Measure RTT with reflector quality filtering.

    Filters ping_hosts through ReflectorScorer active set. Uses
    ping_hosts_with_results for per-host attribution. Records
    success/failure back to scorer for quality tracking.

    Graceful degradation:
        3+ active -> median-of-N
        2 active  -> average-of-2
        1 active  -> single ping
        0 active  -> force best-scoring (handled by get_active_hosts)
    """
    active = self._reflector_scorer.get_active_hosts()
    n = len(active)

    if self.use_median_of_three and n >= 3:
        # Ping up to 3 active hosts concurrently with attribution
        hosts_to_ping = active[:3]
        results = self.rtt_measurement.ping_hosts_with_results(
            hosts=hosts_to_ping, count=1, timeout=3.0
        )
        # Record per-host results for quality scoring
        for host, rtt in results.items():
            self._reflector_scorer.record_result(host, rtt is not None)
        # Persist events if any transitions occurred
        self._persist_reflector_events()

        rtts = [v for v in results.values() if v is not None]
        if len(rtts) >= 2:
            median_rtt = statistics.median(rtts)
            self.logger.debug(
                f"{self.wan_name}: Median-of-{len(rtts)} RTT = {median_rtt:.2f}ms "
                f"(active={n} reflectors)"
            )
            return median_rtt
        elif len(rtts) == 1:
            return rtts[0]
        else:
            self.logger.warning(
                f"{self.wan_name}: All pings failed (median-of-three, active={n})"
            )
            return None
    elif self.use_median_of_three and n == 2:
        # Average of two active hosts
        results = self.rtt_measurement.ping_hosts_with_results(
            hosts=active[:2], count=1, timeout=3.0
        )
        for host, rtt in results.items():
            self._reflector_scorer.record_result(host, rtt is not None)
        self._persist_reflector_events()

        rtts = [v for v in results.values() if v is not None]
        if rtts:
            avg_rtt = statistics.mean(rtts)
            self.logger.debug(
                f"{self.wan_name}: Average-of-{len(rtts)} RTT = {avg_rtt:.2f}ms "
                f"(active=2 reflectors)"
            )
            return avg_rtt
        else:
            self.logger.warning(
                f"{self.wan_name}: All pings failed (average-of-two)"
            )
            return None
    else:
        # Single host ping (n==1, or use_median_of_three is False)
        host = active[0]
        rtt = self.rtt_measurement.ping_host(host, count=1)
        self._reflector_scorer.record_result(host, rtt is not None)
        self._persist_reflector_events()
        return rtt
```

```python
def _persist_reflector_events(self) -> None:
    """Persist any pending reflector deprioritization/recovery events to SQLite.

    Checks ReflectorScorer for transition events and writes them to the
    reflector_events table. Never raises -- follows AlertEngine pattern.
    """
    if self._metrics_writer is None:
        return

    scorer = self._reflector_scorer
    for status in scorer.get_all_statuses():
        # Check if this host just transitioned by comparing with last known state
        # Use a simple approach: track transitions in the scorer itself
        pass

    # NOTE: Actual persistence uses scorer's _pending_events list pattern.
    # ReflectorScorer.record_result appends transition events to _pending_events.
    # This method drains and persists them.
    events = scorer.drain_events()
    for event in events:
        try:
            import json
            import time as time_mod

            timestamp = int(time_mod.time())
            details = json.dumps({
                "host": event["host"],
                "score": round(event["score"], 3),
                "event": event["event_type"],
            })
            self._metrics_writer.connection.execute(
                "INSERT INTO reflector_events "
                "(timestamp, event_type, host, wan_name, score, details) "
                "VALUES (?, ?, ?, ?, ?, ?)",
                (timestamp, event["event_type"], event["host"],
                 self.wan_name, round(event["score"], 3), details),
            )
        except Exception:
            self.logger.warning(
                "Failed to persist reflector event %s for %s",
                event["event_type"], event["host"], exc_info=True,
            )
```
