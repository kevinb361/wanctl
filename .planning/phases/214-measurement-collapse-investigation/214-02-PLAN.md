# scripts/phase214-extract.py
class FlentExtractionError(RuntimeError):
    """Raised when expected flent series is missing or empty (D-10 fail-closed)."""

def extract_flent_latency(path: Path) -> dict:
    """Return:
      {
        "p50_ms": float, "p95_ms": float, "p99_ms": float,
        "min_ms": float, "max_ms": float, "mean_ms": float,
        "sample_count": int,
        "window_start_utc": str,    # ISO8601 from metadata.T0 (e.g. "2026-04-16T08:59:03.512987+00:00")
        "window_end_utc":   str,    # ISO8601, computed as T0 + TOTAL_LENGTH seconds (consumers convert via datetime.fromisoformat when epoch math is needed)
        "ping_command": str | None, # metadata.SERIES_META['Ping (ms) ICMP'].COMMAND
      }
    Raises FlentExtractionError if raw_values['Ping (ms) ICMP'] is missing/empty,
    or if metadata.T0/TIME is missing.
    Reads via gzip.open(path, "rt") + json.load. Computes percentiles by sorting
    and indexing values (n // 2, int(n*0.95), int(n*0.99) with min(n-1, ...) clamp).
    Note (revision 2026-05-27): window_end_utc is an ISO8601 string (NOT epoch float).
    This standardizes the time-key type across phase214 extractor / aligner / matrix-summary,
    resolves checker Warning #5 (asymmetric naming), and matches the live /health NDJSON
    convention which uses `t_wall` (ISO8601) exclusively."""

def extract_flent_throughput(path: Path) -> dict:
    """Return:
      {
        "throughput_median_mbps": float,
        "throughput_p95_mbps": float,
        "throughput_max_mbps": float,
        "sample_count": int,
        "series_key_used": str,     # one of: "TCP download sum"|"TCP totals"|"TCP download avg"
      }
    Raises FlentExtractionError if none of the keys yield a usable list.
    """

# CLI:
#   python3 scripts/phase214-extract.py --flent-gz <path> --output-json <path>
# Exit codes: 0 success; nonzero on FlentExtractionError (message to stderr).

# Verified .flent.gz shape (from 214-RESEARCH.md §Pattern 1, schema inspection):
#   data['raw_values']['Ping (ms) ICMP'] -> list[{'seq': float, 't': unix-float, 'val': float-ms}]  AUTHORITATIVE
#   data['results']['Ping (ms) ICMP']    -> list[float|None] binned to step_size  DO NOT USE
#   data['results']['TCP download sum']  -> list[float|None] Mbit/s per step_size bin
#   data['metadata']['T0']               -> 'YYYY-MM-DDTHH:MM:SS.ffffffZ' UTC
#   data['metadata']['TOTAL_LENGTH']     -> float seconds (includes warmup/cooldown)
#   data['metadata']['SERIES_META']['Ping (ms) ICMP']['COMMAND'] -> ping command string

# Anti-pattern reference (do NOT replicate): scripts/phase213-classify.py:173-191 _flent_summary()
# which looks for a non-existent `flent-summary.json` and falls back to {} → 0.0.
