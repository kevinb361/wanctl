def build_matrix_summary(signal_sheets: list[dict]) -> dict:
    """Aggregate per-window signal-sheet.json dicts into a matrix-summary dict.

    Returns (schema locked to 214-RESEARCH.md §Open Questions Q4):
      {
        "phase": 214,
        "started_utc": str,    # ISO8601, min(window.started_utc); source: per-window signal-sheet.json
                                #   which sources from extract_flent_latency().window_start_utc (ISO8601 str)
        "ended_utc": str,      # ISO8601, max(window.ended_utc); source: extract_flent_latency().window_end_utc
                                #   (ISO8601 str; revision 2026-05-27 — was previously window_end_utc_unix epoch
                                #   float in 214-02; standardized to ISO8601 to resolve checker Warning #5)
        "git_head_sha": str,
        "verdict": "pass" | "fail" | "ambiguous" | "partial",   # "partial" added per Codex HIGH-4
        "primary_driver": str | None,
        "ranked_drivers": list[str],
        "windows": [
          {
            "window": "off-peak" | "daytime" | "prime-time" | "att-contrast",
            "run_dir": str,
            "p50_ms": float, "p95_ms": float, "p99_ms": float,
            "verdict": str,
            "primary_driver": str | None,
            "ranked": list[str],
            "signal_disposition": "form_b" | "form_c" | "none",
          },
          ...
        ],
        "signal_disposition": "form_b" | "form_c" | "none",
        "mutation_posture": "read-only",   # constant for this phase
        "partial_reason": str | None,      # Codex HIGH-4: populated ONLY when verdict="partial"
        "missing_windows": list[str],      # Codex HIGH-4: missing {off-peak,daytime,prime-time}
      }
    """

def matrix_verdict(window_verdicts: list[str], *, full_window_set: bool = True) -> str:
    """Roll-up rule (Codex HIGH-4 — partial verdict added 2026-05-28):
      - any 'fail' in inputs → 'fail'
      - else any 'ambiguous'  → 'ambiguous'
      - else (all 'pass') AND full_window_set is True → 'pass'
      - else (all 'pass') AND full_window_set is False → 'partial'
      Partial NEVER returns 'pass'. The full_window_set flag is set by the CLI based
      on whether the discovered Spectrum set covered {off-peak, daytime, prime-time}.
    """

# with partial_reason copied verbatim. Partial NEVER becomes "pass". Partial

# Primary-driver aggregation across windows: use the most-frequent fired driver across
# windows that had non-"pass" verdict. Tie-break: lexicographic. Document this exact rule

#   1. Phase base SHA: read from `PHASE214_BASE_SHA` env var, else `git rev-parse HEAD~10`

#   2. `git diff --name-only $BASE..HEAD -- src/wanctl/` returns 0 lines (D-14).
#   3. `git diff --name-only $BASE..HEAD -- scripts/phase213-classify.py scripts/phase213-baseline-capture.sh` returns 0 lines (D-11).

# Forbidden mutation-recommendation regex (D-12/D-13/D-14 observational-only invariant):
#
# REVISION 2026-05-28 (Codex MED-10): the previous `(^|\W)` prefix would match
# the regex's own negative test prose like "restart wanctl is a future-phase
# consideration" because `\W` accepts any non-word character including word
# starts inside narrative. Tightened to line-anchored command/assignment forms
# with optional shell-prompt or sudo prefix — the "self-invalidating grep gate"
# pitfall called out in planner critical_rules.
#
#   (?ixm)
#     ^\s*\$?\s*(sudo\s+)?
#     (
#        systemctl\s+restart\s+wanctl(@\S+)?\b
#      | service\s+wanctl(@\S+)?\s+restart\b
#      | restart\s+wanctl(@\S+)?\b
#      | ceiling_mbps\s*[:=]\s*\d
#      | setpoint_mbps\s*[:=]\s*\d
#      | /etc/wanctl/\S+\s+(edit|write|modify)\b
#      | mikrotik\s+(write|set|change)\b
#      | routeros\s+(write|set|change)\b
#      | steering\s+(toggle|enable|disable)\b
#     )
#
# Flags: (?i)+(?x)+(?m). The (?m) flag makes `^` match every line start so
# command forms inside fenced code blocks are caught. The optional prefix
# `^\s*\$?\s*(sudo\s+)?` accepts shell-prompt-style examples like
# `$ sudo systemctl restart wanctl@spectrum`. Narrative like "we considered
# restart wanctl but decided not to" does NOT match — the line does not start
# with the command form. Tests in test_mutation_regex_does_not_false_positive
# pin both halves of this contract.
