    Returns (schema locked to 214-RESEARCH.md §Open Questions Q4):
      {
        "phase": 214,
        "started_utc": str,    # ISO8601, min(window.started_utc); source: per-window signal-sheet.json
                                #   which sources from extract_flent_latency().window_start_utc (ISO8601 str)
        "ended_utc": str,      # ISO8601, max(window.ended_utc); source: extract_flent_latency().window_end_utc
                                #   (ISO8601 str; revision 2026-05-27 — was previously window_end_utc_unix epoch
                                #   float in 214-02; standardized to ISO8601 to resolve checker Warning #5)
        "git_head_sha": str,
        "verdict": "pass" | "fail" | "ambiguous",
        "primary_driver": str | None,
        "ranked_drivers": list[str],
        "windows": [
          {
            "window": "off-peak" | "daytime" | "prime-time" | "att-contrast",
            "run_dir": str,
            "p50_ms": float, "p95_ms": float, "p99_ms": float,
            "verdict": str,
            "primary_driver": str | None,
            "ranked": list[str],
            "signal_disposition": "form_b" | "form_c" | "none",
          },
          ...
        ],
        "signal_disposition": "form_b" | "form_c" | "none",
        "mutation_posture": "read-only",   # constant for this phase
      }
    """

def matrix_verdict(window_verdicts: list[str]) -> str:

      - any 'fail' in inputs → 'fail'
      - else any 'ambiguous'  → 'ambiguous'
      - else (all 'pass')     → 'pass'
    """
