def classify(aligned_rows: list[dict]) -> dict:
    """Apply all six drivers; return {primary, ranked, drivers, evidence_summary}.

    Returns:
      {
        "primary_driver": str | None,    # "reflector_loss" | "icmp_udp_divergence" | "stale_cached_rtt"
                                          #  | "steering_behavior" | "cake_queue_mismatch" | "external_path"
                                          #  | None
        "ranked": list[str],              # ordered by evidence_strength_score desc
        "drivers": {
            "<name>": {"fired": bool, "evidence": str, "score": int, "first_unix": int|None}
        },
      }

    primary_driver contract (revision 2026-05-27 ŌĆö resolves checker Warning #4):
      - If aligned_rows (filtered to in_flent_window) is EMPTY ŌåÆ primary_driver = None
        (degenerate: no evidence to classify).
      - If aligned_rows is non-empty AND Ōēź1 driver fired ŌåÆ primary_driver = ranked[0]
        (highest-score driver among 1..5; lexicographic tiebreak).
      - If aligned_rows is non-empty AND NO driver fired ŌåÆ primary_driver = "external_path"
        (catch-all per Pattern 4 driver 6).
      The `None` and `external_path` cases are mutually exclusive ŌĆö `None` means "no rows
      to evaluate", `external_path` means "evaluated rows, no signal driver matched".
    """

def verdict_for_window(latency: dict, drivers: dict) -> str:
    """Returns 'pass', 'fail', or 'ambiguous' per D-06 boundaries.

    Pass: latency.p99_ms < 500 AND drivers['reflector_loss']['fired'] is False
                                AND drivers['icmp_udp_divergence']['fired'] is False
    Fail: latency.p99_ms > 1000 AND any(d['fired'] for d in drivers.values())
    Otherwise: 'ambiguous'
    """
