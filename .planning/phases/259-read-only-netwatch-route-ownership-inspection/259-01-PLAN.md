class RouteOwnershipClient(Protocol):
    def run_cmd(self, cmd: str, capture: bool = False,
                timeout: int | None = None) -> tuple[int, str, str]: ...

class RouteOwnershipGuard:
    NETWATCH_PRINT = "/tool netwatch print detail"
    SCRIPT_PRINT = "/system script print detail"
    def __init__(self, router_client: RouteOwnershipClient) -> None: ...
    def inspect(self) -> RouteOwnershipGuardResult: ...

@dataclass(frozen=True)
class RouteOwnershipGuardResult:
    status: Literal["ok", "conflict", "error"]
    active_allowed: bool
    owner: str
    conflicts: tuple[RouteOwnershipConflict, ...] = ()
    error: str | None = None

@dataclass(frozen=True)
class RouteOwnershipConflict:
    source: str       # "netwatch" for netwatch-sourced conflicts
    name: str
    script: str | None
    reason: str

Create `src/wanctl/steering/route_ownership_inspector.py`. Mirror module/class structure of
route_ownership_guard.py. Implement `class RouteOwnershipInspector`:

__init__(self, router_client, route_manager, *, interval_sec=60.0, logger=None):
  - self._guard = RouteOwnershipGuard(router_client)  # dedicated, UNCONDITIONAL (D2) — NEVER reuse
    daemon.route_ownership_guard, which is None when route_management disabled/off.
  - store router_client, route_manager, interval, logger; threading.Lock; threading.Event (stop);
    self._thread=None; self._cached = self._compute() is NOT called here — instead seed
    self._cached with an initial "starting" snapshot (inspector_status="starting", observed_owner
    "unknown", empty netwatch/routes, last_inspected_at None) so snapshot() is serveable before
    first refresh.

start(self): run one synchronous refresh() to populate the cache, then spawn
  threading.Thread(target=self._loop, daemon=True, name="ownership-inspection") and start it.
stop(self): self._stop.set(); if thread: thread.join(timeout=5.0).
_loop(self): while not self._stop.is_set(): if self._stop.wait(self._interval): break; self.refresh().
snapshot(self): with self._lock: return dict(self._cached).
refresh(self): result = self._compute()  (ALL I/O outside the lock); with self._lock: self._cached = result.

_compute(self) -> dict — wrap the whole body in try/except Exception (D4 fail-open). Inside:
  - guard_result = self._guard.inspect()
  - read netwatch list directly for the count: rc,out,err =
    router_client.run_cmd(RouteOwnershipGuard.NETWATCH_PRINT, capture=True, timeout=5); parse JSON list
    (reuse the guard's defensive list-parse idiom: dict→[dict], non-list→treat as error). entries_count =
    len(list). NOTE: this is the second netwatch read per RESEARCH A3 "read once or twice" — one extra
    GET at 60s cadence is acceptable and avoids touching the guard contract.
  - route_mutating_active_count = len([c for c in guard_result.conflicts if c.source == "netwatch"]).
  - observed_owner via the D3 helper _observed_owner(guard_result, route_mode).
  - rm_snap = route_manager.status_snapshot(); configured_owner = str(rm_snap.get("active_owner","unknown"));
    route_mode = str(rm_snap.get("mode","off")).
  - read /ip route print (run_cmd("/ip route print", capture=True, timeout=5)); parse list; total =
    len(list); default_routes = projection of dst-address=="0.0.0.0/0" rows using module-level
    _routeros_bool (copy the idiom from route_manager.py:433 — do NOT import private helpers across
    modules; define a local _routeros_bool and _coerce_int) for disabled/distance.
  - inspector_status = "error" if guard_result.status=="error" else "ok". If the netwatch/route reads
    themselves fail (rc!=0 or parse error) raise inside the try so the except path sets error state.
  - match = (observed_owner == configured_owner) and observed_owner != "unknown".
  - last_inspected_at = datetime.now(UTC).isoformat().
  - return the full pinned dict shape: observed_owner, configured_owner, match, inspector_status,
    inspector_error (None on success), last_inspected_at, netwatch{entries_count,
    route_mutating_active_count}, routes{total_route_count, default_routes:[...]}.
  except branch: build and return a complete fail-open dict (inspector_status="error",
    observed_owner="unknown", match=False, inspector_error=str(exc), netwatch/routes zeroed/empty,
    last_inspected_at=datetime.now(UTC).isoformat()).

Keep new logic ~120 LoC. No RouterOS mutation, no command interpolation of router data.
