    ```bash
    #!/usr/bin/env bash
    # Phase 192 soak capture — D-08 three-category evidence.
    #
    # Portable controller compliance (CLAUDE.md): all deployment-specific
    # values are read from environment variables. This script ships with NO
    # hardcoded IPs, hostnames, or WAN names. See
    # scripts/phase192-soak-capture.env.example for the full required-var list.
    #
    # Usage:
    #   WANS="spectrum att" \
    #   WANCTL_SPECTRUM_HEALTH_URL=... WANCTL_SPECTRUM_SSH_HOST=... \
    #   WANCTL_ATT_HEALTH_URL=...      WANCTL_ATT_SSH_HOST=... \
    #     scripts/phase192-soak-capture.sh {pre|post}
    set -euo pipefail

    phase_window="${1:?usage: $0 {pre|post}}"
    case "$phase_window" in
      pre|post) ;;
      *) echo "first arg must be 'pre' or 'post', got: $phase_window" >&2; exit 2 ;;
    esac

    # Required: WANS (space-separated WAN names).
    if [ -z "${WANS:-}" ]; then
      echo "ERROR: WANS env var is required (space-separated WAN names, e.g. 'spectrum att')" >&2
      echo "  See scripts/phase192-soak-capture.env.example for the full required-var list." >&2
      exit 2
    fi

    # For each WAN, require HEALTH_URL and SSH_HOST. Fail fast with a clear
    # message if any is missing — do NOT fall back to any default.
    for wan in $WANS; do
      upper=$(echo "$wan" | tr '[:lower:]' '[:upper:]')
      health_var="WANCTL_${upper}_HEALTH_URL"
      ssh_var="WANCTL_${upper}_SSH_HOST"
      if [ -z "${!health_var:-}" ]; then
        echo "ERROR: $health_var is required (health endpoint URL for WAN '$wan')" >&2
        exit 2
      fi
      if [ -z "${!ssh_var:-}" ]; then
        echo "ERROR: $ssh_var is required (SSH target for WAN '$wan' journalctl reads)" >&2
        exit 2
      fi
    done

    script_dir="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    repo_root="$(cd "$script_dir/.." && pwd)"
    out_dir="$repo_root/.planning/phases/192-reflector-scorer-blackout-awareness-and-log-hygiene/soak/$phase_window"
    mkdir -p "$out_dir"

    capture_ts="$(date -u +%Y-%m-%dT%H:%M:%SZ)"

    capture_wan() {
      local wan="$1"
      local upper
      upper=$(echo "$wan" | tr '[:lower:]' '[:upper:]')
      local health_url_var="WANCTL_${upper}_HEALTH_URL"
      local ssh_host_var="WANCTL_${upper}_SSH_HOST"
      local health_url="${!health_url_var}"
      local ssh_host="${!ssh_host_var}"

      local raw_dir="$out_dir/${wan}-raw"
      mkdir -p "$raw_dir"
      local raw_health="$raw_dir/health-${capture_ts}.json"
      local raw_journal="$raw_dir/journal-${capture_ts}.log"
      local raw_fusion="$raw_dir/fusion-transitions-${capture_ts}.log"
      local parsed_file="$out_dir/${wan}.json"

      # 1. Fetch /health
      curl --fail --silent --show-error --max-time 5 "$health_url" > "$raw_health"

      local dwell_bypass burst_dl burst_ul
      dwell_bypass=$(jq -r '.download.hysteresis.dwell_bypassed_count // 0' "$raw_health")
      burst_dl=$(jq -r '.download.burst.trigger_count // 0' "$raw_health")
      burst_ul=$(jq -r '.upload.burst.trigger_count // 0' "$raw_health")

      # 2. Fetch journalctl once, reuse for both fusion and protocol-deprio counts.
      ssh "$ssh_host" \
        "sudo journalctl -u 'wanctl@${wan}' --since '-24h' --no-pager" \
        > "$raw_journal" 2>/dev/null || true

      grep -E 'Fusion healer.*->' "$raw_journal" > "$raw_fusion" || true
      local fusion_count protocol_count
      fusion_count=$(wc -l < "$raw_fusion" | tr -d ' ')
      protocol_count=$(grep -c 'Protocol deprioritization detected' "$raw_journal" || echo 0)

      # 3. Assemble the single {wan}.json object (OVERWRITES — idempotent).
      jq -n \
        --arg ts "$capture_ts" \
        --argjson dwell_bypassed_count "$dwell_bypass" \
        --argjson burst_trigger_count_dl "$burst_dl" \
        --argjson burst_trigger_count_ul "$burst_ul" \
        --argjson fusion_transition_count_24h "$fusion_count" \
        --argjson protocol_deprio_count_24h "$protocol_count" \
        '{
          captured_at_iso: $ts,
          dwell_bypassed_count: $dwell_bypassed_count,
          burst_trigger_count_dl: $burst_trigger_count_dl,
          burst_trigger_count_ul: $burst_trigger_count_ul,
          fusion_transition_count_24h: $fusion_transition_count_24h,
          protocol_deprio_count_24h: $protocol_deprio_count_24h
        }' > "$parsed_file"
    }

    for wan in $WANS; do
      capture_wan "$wan"
    done

    echo "Phase 192 soak capture ($phase_window) complete: $out_dir"
    ```
