       if active_layer is RESPONSE_LAYER:

           from wanctl.tuning.strategies.response import RESPONSE_PARAMS as _RESP_PARAMS
           try:
               osc_metrics = _query_wan_metrics(db_path, wc.wan_name, tuning_config.lookback_hours)
               state_by_ts: dict[int, float] = {}
               for row in osc_metrics:
                   if row.get("metric_name") == "wanctl_state":
                       state_by_ts[row["timestamp"]] = row["value"]
               sorted_ts = sorted(state_by_ts.keys())
               # Count transitions in last 60 minutes
               if len(sorted_ts) >= 2:
                   cutoff = sorted_ts[-1] - 3600
                   recent = [ts for ts in sorted_ts if ts >= cutoff]
                   transitions = sum(
                       1 for i in range(1, len(recent))
                       if state_by_ts[recent[i]] != state_by_ts[recent[i - 1]]
                   )
                   span_min = max(1, (recent[-1] - recent[0]) / 60)
                   trans_per_min = transitions / span_min
                   osc_threshold = getattr(wc, '_oscillation_threshold', DEFAULT_OSCILLATION_THRESHOLD)
                   if trans_per_min > osc_threshold:
                       # Lock ALL response parameters
                       for p in _RESP_PARAMS:
                           lock_parameter(wc._parameter_locks, p, OSCILLATION_LOCKOUT_SEC)
                       wan_info["logger"].warning(
                           "[TUNING] %s: oscillation lockout triggered (%.2f trans/min > %.2f), "
                           "all response params locked for %ds",
                           wc.wan_name, trans_per_min, osc_threshold, OSCILLATION_LOCKOUT_SEC,
                       )
                       # Fire Discord alert
                       if hasattr(wc, '_alert_engine') and wc._alert_engine is not None:
                           try:

                                   alert_type="oscillation_lockout",
                                   severity="warning",
