    # (Codex Cycle 3 MEDIUM: derive from ANY contributing replicate, not just latest.
    #  A defect cell with even one BGP-flagged replicate is excluded under D-10.)
    EVIDENCE_ROOT = Path('.planning/phases/220-matrix-runner-scope-a1/evidence')

    def mtr_post_flag_or_across_replicates(cell_id):
        for p in EVIDENCE_ROOT.rglob('phase220-cell.json'):
            try:
                m = json.loads(p.read_text())
            except Exception:
                continue
            if m.get('schema_version') != 1 or not m.get('cell_id'):
                continue
            if re.sub(r'__r\d+$', '', m['cell_id']) == cell_id and bool(m.get('path_change_detected')):

    bgp_excluded = sorted([
        cid for cid, c in per_cell.items()
        if c.get('verdict') == 'cell_defect' and mtr_post_flag_or_across_replicates(cid)

    excluded_set = set(bgp_excluded)

    # ---- Step 0f: re-invoke aggregator's matrix_verdict() on per_cell MINUS excluded (single source of truth)
    # NB: matrix_verdict() takes a LIST of cell dicts (not a dict keyed by cell_id), so convert.

            f"between invocations OR Plan 03 passed it a malformed cell list."
        )
