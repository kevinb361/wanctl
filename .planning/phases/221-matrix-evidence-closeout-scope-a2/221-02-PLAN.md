  VALID_SIDECARS = []
  QUARANTINED = []
  for sidecar in glob(".planning/phases/220-matrix-runner-scope-a1/evidence/**/phase220-cell.json"):
      manifest = json.load(sidecar)
      if manifest.get("schema_version") != 1 or not manifest.get("cell_id"):
          QUARANTINED.append(str(sidecar.parent.relative_to(repo_root)))
          continue
      manifest["__sidecar_path"] = str(sidecar.relative_to(repo_root))
      manifest["__sidecar_parent_canonical"] = str(Path(sidecar.parent).resolve())
      VALID_SIDECARS.append(manifest)

  # DEDUPLICATE on (base_cell_id, replicate_index, run_dir_canonical) tuple
  # (Resolves Codex Cycle 2 HIGH #2: rehearsal dir + RUN-* dir produce two manifests
  #  for the same r1 replicate; counting both inflates dallas__spectrum__daytime to 2/3.)
  DEDUP = {}  # key: (base, replicate_index, run_dir_canonical) -> manifest
  DUPLICATE_SIDECARS = []  # audit list of dropped sidecar paths
  for m in VALID_SIDECARS:
      base = re.sub(r"__r\d+$", "", m["cell_id"])
      rep_idx = int(m.get("replicate_index") or 0)
      run_dir_raw = m.get("run_dir") or m["__sidecar_parent_canonical"]
      run_dir_canonical = str(Path(run_dir_raw).resolve())
      key = (base, rep_idx, run_dir_canonical)
      existing = DEDUP.get(key)
      if existing is None:
          DEDUP[key] = m
      else:
          # Keep the manifest with the LATEST started_utc as authoritative;
          # record the dropped one for audit.
          if (m.get("started_utc") or "") > (existing.get("started_utc") or ""):
              DUPLICATE_SIDECARS.append(existing["__sidecar_path"])
              DEDUP[key] = m
          else:
              DUPLICATE_SIDECARS.append(m["__sidecar_path"])

  # Group dedup'd manifests by base cell_id
  by_base_cell = defaultdict(list)
  for m in DEDUP.values():
      base = re.sub(r"__r\d+$", "", m["cell_id"])
      by_base_cell[base].append(m)

  for cell in yaml.load(scripts/phase220-matrix.yaml)["cells"]:
      replicates = by_base_cell.get(cell["cell_id"], [])
      valid_n = len(replicates)
      attempts_annotation = parse_attempts_from_ledger_notes(cell["cell_id"])
      if attempts_annotation >= 3 and valid_n < 3:
          status = "incomplete"
      elif valid_n == 3:
          status = "complete"
      elif 1 <= valid_n <= 2:
          status = "partial"
      else:
          status = "pending"
      latest = max(replicates, key=lambda m: m["started_utc"]) if replicates else None
      ledger_row[cell["cell_id"]] = {
          "replicates": f"{valid_n}/3",
          "last_replicate_utc": latest["started_utc"] if latest else "—",
          "mtr_pre_sha256": short_sha256(latest_pre_mtr_path) if latest else "—",
          "mtr_post_flag": bool(latest["path_change_detected"]) if latest else "—",
          "base_sha": latest["base_sha"] if latest else "—",
          "status": status,
          # notes column preserved + operator may add attempts:<N> annotation
      }
  frontmatter.completed_replicates = sum(min(len(v), 3) for v in by_base_cell.values()
                                          if k_is_in_18_cell_list(v))
  frontmatter.canonical_complete = count(rows where target=="dallas" and status=="complete")
  frontmatter.supplemental_incomplete = count(rows where target in supplemental and status=="incomplete")
  frontmatter.quarantined_run_dirs = QUARANTINED
  frontmatter.duplicate_sidecars = DUPLICATE_SIDECARS  # audit list (Cycle 2 HIGH #2)
  frontmatter.last_session_utc = now_utc_iso8601()
