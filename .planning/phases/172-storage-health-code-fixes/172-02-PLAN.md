**Step 1: Create `src/wanctl/storage/db_utils.py`**
New module for shared DB discovery and multi-DB query logic:

```python
"""Per-WAN database auto-discovery utilities.

Supports the per-WAN DB split (D-05) by providing auto-discovery
of metrics-*.db files for CLI tools that need unified output.

Merge semantics (addresses review feedback):
- Results from all discovered DBs are concatenated and sorted by timestamp
- No deduplication: each per-WAN DB contains only that WAN's data, so
  overlapping timestamps from different WANs are distinct rows (wan_name differs)
- Partial failures: if a DB is unreadable/corrupt, log a warning and continue
  with results from remaining DBs
- Legacy coexistence: if per-WAN files exist (metrics-*.db), the legacy
  metrics.db is IGNORED even if it exists. This prevents double-counting
  during the transition period after deploy.
"""

import logging
from pathlib import Path

from wanctl.storage.writer import DEFAULT_DB_PATH

logger = logging.getLogger(__name__)

# Directory containing per-WAN DB files
WAN_DB_DIR = DEFAULT_DB_PATH.parent  # /var/lib/wanctl
WAN_DB_GLOB = "metrics-*.db"


def discover_wan_dbs(db_dir: Path | None = None) -> list[Path]:
    """Auto-discover per-WAN metrics DB files with legacy fallback.

    Precedence rules:
    1. If metrics-*.db files exist in db_dir, return them (sorted).
       Legacy metrics.db is IGNORED to prevent double-counting.
    2. If NO per-WAN files exist but legacy metrics.db exists, return [legacy].
    3. If no DB files exist at all, return [].

    Args:
        db_dir: Directory to search. Defaults to /var/lib/wanctl.

    Returns:
        Sorted list of Path objects to DB files.
    """
    search_dir = db_dir or WAN_DB_DIR
    dbs = sorted(search_dir.glob(WAN_DB_GLOB))
    if dbs:
        # Per-WAN files found -- ignore legacy DB even if it exists
        return dbs
    # No per-WAN files -- fall back to legacy shared DB
    legacy = search_dir / "metrics.db"
    if legacy.exists():
        return [legacy]
    return []


def query_all_wans(
    query_fn,
    db_paths: list[Path] | None = None,
    **kwargs,
) -> list[dict]:
    """Query across all per-WAN DBs and merge results.

    Merge semantics:
    - Concatenates results from all DBs
    - Sorts by timestamp for chronological output
    - Each row retains its wan_name field for source identification
    - On partial failure (one DB unreadable), logs warning and continues

    Args:
        query_fn: A callable accepting db_path as first arg (e.g., query_metrics,
                  query_tuning_params, query_alerts).
        db_paths: Explicit list of DB paths. If None, uses discover_wan_dbs().
        **kwargs: Forwarded to query_fn.

    Returns:
        Merged list of dicts sorted by timestamp.
    """
    paths = db_paths if db_paths is not None else discover_wan_dbs()
    results = []
    for db in paths:
        try:
            rows = query_fn(db_path=db, **kwargs)
            results.extend(rows)
        except Exception as e:
            logger.warning("Failed to query %s, skipping: %s", db.name, e)
    # Sort by timestamp for chronological merged output
    results.sort(key=lambda r: r.get("timestamp", 0))
    return results
```
