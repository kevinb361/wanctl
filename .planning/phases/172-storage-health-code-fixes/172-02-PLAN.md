**Step 1: Create `src/wanctl/storage/db_utils.py`**
New module for shared DB discovery logic:

```python
"""Per-WAN database auto-discovery utilities.

Supports the per-WAN DB split (D-05) by providing auto-discovery
of metrics-*.db files for CLI tools that need unified output.
"""

from pathlib import Path

from wanctl.storage.writer import DEFAULT_DB_PATH

# Directory containing per-WAN DB files
WAN_DB_DIR = DEFAULT_DB_PATH.parent  # /var/lib/wanctl
WAN_DB_GLOB = "metrics-*.db"


def discover_wan_dbs(db_dir: Path | None = None) -> list[Path]:
    """Auto-discover per-WAN metrics DB files with legacy fallback.

    Searches for metrics-*.db files in the given directory.
    Falls back to the legacy shared metrics.db if no per-WAN files found.
    Returns empty list if no DB files exist at all.

    Args:
        db_dir: Directory to search. Defaults to /var/lib/wanctl.

    Returns:
        Sorted list of Path objects to DB files.
    """
    search_dir = db_dir or WAN_DB_DIR
    dbs = sorted(search_dir.glob(WAN_DB_GLOB))
    if not dbs:
        legacy = search_dir / "metrics.db"
        if legacy.exists():
            return [legacy]
    return dbs
```
