    Step 2 — Author `scripts/phase200-saturation-canary.sh`. Use `#!/usr/bin/env bash` + `set -euo pipefail`. Follow the phase196 require_var/mkdir pattern. Skeleton (operator may refine layout but MUST preserve every gate listed below):

    ```bash
    #!/usr/bin/env bash
    # Phase 200 saturation canary — VALN-06 deploy gate.
    #
    # Runs a saturated iperf3 -P4 UL loop at the deployed UL ceiling on Spectrum,
    # samples /health 1Hz throughout, fails closed if the UL controller collapses
    # to floor in ANY cycle. Bookended by 60s pre/post idle baselines.
    #
    # Per D-07: this is the PRIMARY deploy gate. Plan 06 reads verdict.json and
    # proceeds with the production deploy only on verdict=pass. The 24h soak
    # (Plan 07) is a regression watchdog, not the verdict.
    #
    # Per D-10: rollback protocol is predefined — revert /opt/wanctl binary to
    # v1.40. The verdict.json records the rollback command path for fast recovery.
    #
    # Usage:
    #   source scripts/phase200-saturation-canary.env  # operator-private
    #   scripts/phase200-saturation-canary.sh
    #   scripts/phase200-saturation-canary.sh --help
    #
    # Exit codes:
    #   0 = PASS (no UL collapse to floor; verdict.json verdict="pass")
    #   1 = FAIL (UL collapsed to floor in >=1 cycle; verdict.json verdict="fail"; revert binary per D-10)
    #   2 = ABORT (env validation, /health unreachable, iperf3 unreachable; verdict.json verdict="abort"; do NOT deploy)

    set -euo pipefail

    print_help() {
        cat <<'HELP'
    Phase 200 saturation canary (VALN-06).

    Required env vars (set before running, e.g. via `source phase200-saturation-canary.env`):
      PHASE200_OUT_DIR              Repo-local evidence directory.
      PHASE200_SPECTRUM_HEALTH_URL  /health endpoint (Spectrum).
      PHASE200_IPERF_TARGET         iperf3 server hostname or IP.
      PHASE200_IPERF_LOCAL_BIND     Local source IP that exits Spectrum.

    Optional overrides:
      PHASE200_LOAD_DURATION_SEC    Default 900 (15 min).
      PHASE200_BASELINE_DURATION_SEC Default 60.
      PHASE200_HEALTH_POLL_HZ        Default 1 (1 sample/sec).

    Output: ${PHASE200_OUT_DIR}/canary/<UTC-TS>/
      pre_idle_baseline.json
      loaded_capture.ndjson
      loaded_iperf_summary.json
      post_idle_baseline.json
      verdict.json

    See plan 200-05 for full gate semantics.
    HELP
    }

    [[ "${1:-}" == "--help" || "${1:-}" == "-h" ]] && { print_help; exit 0; }

    require_var() {
        local name="$1"
        if [[ -z "${!name:-}" ]]; then
            echo "ABORT: env var $name is not set. See $0 --help." >&2
            exit 2
        fi
    }

    require_var PHASE200_OUT_DIR
    require_var PHASE200_SPECTRUM_HEALTH_URL
    require_var PHASE200_IPERF_TARGET
    require_var PHASE200_IPERF_LOCAL_BIND

    LOAD_DURATION_SEC="${PHASE200_LOAD_DURATION_SEC:-900}"
    BASELINE_DURATION_SEC="${PHASE200_BASELINE_DURATION_SEC:-60}"
    HEALTH_POLL_HZ="${PHASE200_HEALTH_POLL_HZ:-1}"
    POLL_INTERVAL_SEC="$(awk "BEGIN{print 1/${HEALTH_POLL_HZ}}")"

    UTC_TS="$(date -u +%Y%m%dT%H%M%SZ)"
    RUN_DIR="${PHASE200_OUT_DIR}/canary/${UTC_TS}"
    mkdir -p "$RUN_DIR"

    PRE_BASELINE="${RUN_DIR}/pre_idle_baseline.json"
    LOADED_CAPTURE="${RUN_DIR}/loaded_capture.ndjson"
    LOADED_IPERF="${RUN_DIR}/loaded_iperf_summary.json"
    POST_BASELINE="${RUN_DIR}/post_idle_baseline.json"
    VERDICT="${RUN_DIR}/verdict.json"

    # ABORT-on-tooling-missing
    for tool in jq curl iperf3; do
        command -v "$tool" >/dev/null || { echo "ABORT: $tool not on PATH" >&2; exit 2; }
    done

    # ABORT if /health is unreachable BEFORE we start the run
    if ! curl --silent --fail --max-time 5 "$PHASE200_SPECTRUM_HEALTH_URL" \
            | jq -e '.wans[0].upload | has("current_rate_mbps") and has("floor_mbps")' >/dev/null; then
        echo "ABORT: $PHASE200_SPECTRUM_HEALTH_URL did not return expected /health shape" >&2
        jq -n --arg run_id "$UTC_TS" '{phase:200, run_id:$run_id, verdict:"abort", reason:"health_unreachable_or_shape_invalid"}' \
            > "$VERDICT"
        exit 2
    fi

    # ABORT if iperf3 server is unreachable (1s probe)
    if ! iperf3 -c "$PHASE200_IPERF_TARGET" -B "$PHASE200_IPERF_LOCAL_BIND" -t 1 -J >/dev/null 2>&1; then
        echo "ABORT: iperf3 -c $PHASE200_IPERF_TARGET -B $PHASE200_IPERF_LOCAL_BIND probe failed" >&2
        jq -n --arg run_id "$UTC_TS" '{phase:200, run_id:$run_id, verdict:"abort", reason:"iperf_target_unreachable"}' \
            > "$VERDICT"
        exit 2
    fi

    # Helper: capture N seconds of /health at 1Hz, write to NDJSON file. Each line is one sample.
    capture_health() {
        local out="$1"
        local duration="$2"
        local samples=0
        local end_ts=$(( $(date +%s) + duration ))
        > "$out"
        while [[ "$(date +%s)" -lt "$end_ts" ]]; do
            curl --silent --max-time 2 "$PHASE200_SPECTRUM_HEALTH_URL" \
                | jq -c --arg sampled_at "$(date -u +%Y-%m-%dT%H:%M:%SZ)" '. + {sampled_at_utc: $sampled_at}' \
                >> "$out" || true
            samples=$((samples + 1))
            sleep "$POLL_INTERVAL_SEC"
        done
        echo "$samples"
    }

    summarize_baseline() {
        # input ndjson, output json blob with min/p50/p99 baseline_rtt_ms + sample_count
        local in_ndjson="$1"
        local out_json="$2"
        jq -s '
            map(.wans[0].rtt.baseline_rtt_ms // .wans[0].upload.current_rate_mbps) as $rtts
            | {
                sample_count: length,
                baseline_rtt_ms: {
                    min: ([.[].wans[0].rtt.baseline_rtt_ms // null] | map(select(. != null)) | min),
                    p50: ([.[].wans[0].rtt.baseline_rtt_ms // null] | map(select(. != null)) | sort | .[length/2|floor]),
                    max: ([.[].wans[0].rtt.baseline_rtt_ms // null] | map(select(. != null)) | max)
                }
            }
        ' "$in_ndjson" > "$out_json"
    }

    echo "[$(date -u +%FT%TZ)] Pre-idle baseline ($BASELINE_DURATION_SEC s) ..."
    PRE_NDJSON="${RUN_DIR}/pre_idle_baseline.ndjson"
    capture_health "$PRE_NDJSON" "$BASELINE_DURATION_SEC" >/dev/null
    summarize_baseline "$PRE_NDJSON" "$PRE_BASELINE"
    PRE_BASELINE_RTT_P50="$(jq -r '.baseline_rtt_ms.p50' "$PRE_BASELINE")"

    # Capture an unambiguous start-of-loaded-window UTC timestamp BEFORE iperf3
    # launches. The verdict.json `started_at_utc` field uses this captured value
    # rather than fragile inline `date -u -d "<seconds ago>" 2>/dev/null || ...`
    # subtraction at the end of the run.
    LOADED_START_UTC="$(date -u +%FT%TZ)"
    echo "[$LOADED_START_UTC] Loaded window ($LOAD_DURATION_SEC s) — iperf3 -P4 saturated UL ..."
    # Run iperf3 in foreground? No — operator wants concurrent /health sampling.
    # Run iperf3 in background, capture summary at end, while the foreground samples /health.
    iperf3 -c "$PHASE200_IPERF_TARGET" -B "$PHASE200_IPERF_LOCAL_BIND" \
           -P 4 -t "$LOAD_DURATION_SEC" -J > "$LOADED_IPERF" &
    IPERF_PID=$!
    capture_health "$LOADED_CAPTURE" "$LOAD_DURATION_SEC" >/dev/null
    wait "$IPERF_PID" || echo "WARN: iperf3 wait returned non-zero (continuing — verdict is /health-driven)" >&2

    echo "[$(date -u +%FT%TZ)] Post-idle baseline ($BASELINE_DURATION_SEC s) ..."
    POST_NDJSON="${RUN_DIR}/post_idle_baseline.ndjson"
    capture_health "$POST_NDJSON" "$BASELINE_DURATION_SEC" >/dev/null
    summarize_baseline "$POST_NDJSON" "$POST_BASELINE"
    POST_BASELINE_RTT_P50="$(jq -r '.baseline_rtt_ms.p50' "$POST_BASELINE")"

    # Verdict computation — UL floor-hit count during loaded window.
    UL_FLOOR_HITS="$(jq -s '
        map(select(.wans[0].upload.current_rate_mbps == .wans[0].upload.floor_mbps)) | length
    ' "$LOADED_CAPTURE")"
    UL_HYSTERESIS_LAST="$(jq -s '.[-1].wans[0].upload.hysteresis.suppressions_last_60s // null' "$LOADED_CAPTURE")"

    if [[ "$UL_FLOOR_HITS" -gt 0 ]]; then
        VERDICT_VAL="fail"
    else
        VERDICT_VAL="pass"
    fi

    jq -n \
        --arg phase "200" \
        --arg run_id "$UTC_TS" \
        --arg started_at_utc "$LOADED_START_UTC" \
        --arg finished_at_utc "$(date -u +%FT%TZ)" \
        --argjson duration_sec "$((BASELINE_DURATION_SEC * 2 + LOAD_DURATION_SEC))" \
        --argjson ul_floor_hits "$UL_FLOOR_HITS" \
        --argjson pre_rtt_p50 "${PRE_BASELINE_RTT_P50:-null}" \
        --argjson post_rtt_p50 "${POST_BASELINE_RTT_P50:-null}" \
        --argjson ul_hysteresis_60s "${UL_HYSTERESIS_LAST:-null}" \
        --arg verdict "$VERDICT_VAL" \
        --arg rollback_protocol "Per D-10: revert /opt/wanctl to v1.40 binary on cake-shaper. Recommended: keep v1.40 release tarball at /opt/wanctl-v1.40.tar.gz and restore via 'sudo tar -xzf /opt/wanctl-v1.40.tar.gz -C /opt/wanctl && sudo systemctl restart wanctl@spectrum.service'. /etc/wanctl/spectrum.yaml stays in place — pre-Plan-01 binary silently no-ops the new keys." \
        '{
          phase: ($phase | tonumber),
          run_id: $run_id,
          started_at_utc: $started_at_utc,
          finished_at_utc: $finished_at_utc,
          duration_sec: $duration_sec,
          ul_floor_hits_during_load: $ul_floor_hits,
          ul_floor_threshold_hit: ($ul_floor_hits > 0),
          pre_baseline_rtt_ms_p50: $pre_rtt_p50,
          post_baseline_rtt_ms_p50: $post_rtt_p50,
          ul_hysteresis_suppressions_last_60s_at_end: $ul_hysteresis_60s,
          verdict: $verdict,
          rollback_protocol_recorded: true,
          rollback_protocol: $rollback_protocol
        }' > "$VERDICT"

    cat "$VERDICT"

    if [[ "$VERDICT_VAL" == "pass" ]]; then
        echo "[$(date -u +%FT%TZ)] Canary PASS — Plan 06 may proceed with deploy."
        exit 0
    else
        echo "[$(date -u +%FT%TZ)] Canary FAIL — UL collapsed to floor $UL_FLOOR_HITS times. Per D-10: revert /opt/wanctl to v1.40." >&2
        exit 1
    fi
    ```
