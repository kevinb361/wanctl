    6. def mann_whitney_u(x: list[float], y: list[float], *, continuity_correction: bool = False) -> dict
       Implements R3 algorithm:
       - Edge cases: if len(x)==0 or len(y)==0 → return {"u_x": None, "u_y": None, "z": None, "p": None, "degenerate": True, "tie_correction": "wilcoxon-mid-rank"}
       - All-identical pooled samples → {"degenerate": True, ...} (sigma^2 == 0 guard)
       - len(x)==1 and len(y)==1 → degenerate
       - Rank with mid-rank tie correction (assign average rank to tied values)
       - U_x = sum(rank_x) - n_x * (n_x + 1) / 2; U_y = n_x * n_y - U_x
       - mu = n_x * n_y / 2
       - sigma^2 = (n_x * n_y / 12) * ((N + 1) - sum(t^3 - t) / (N * (N - 1)))  where t ranges over tie-group sizes ≥ 2
       - z = (U_x - mu) / sqrt(sigma^2); if continuity_correction: shift U_x by ±0.5 toward mu before computing z
       - p = 2 * (1 - Phi(|z|)) where Phi(z) = 0.5 * (1 + math.erf(z / math.sqrt(2)))
       - Return all numeric fields rounded internally to no fewer than 8 decimal places (Python float native precision; do not truncate the value)

    7. def bootstrap_ci_median_difference(x: list[float], y: list[float], *, seed: int = 220, B: int = 2000, alpha: float = 0.05) -> dict
       Implements R4 algorithm:
       - Edge cases: empty arms → degenerate
       - rng = random.Random(seed)
       - For b in range(B): xs_b = [rng.choice(x) for _ in range(len(x))]; ys_b = [rng.choice(y) for _ in range(len(y))]; diffs.append(statistics.median(xs_b) - statistics.median(ys_b))
       - sorted_diffs = sorted(diffs)
       - lower_idx = int((alpha/2) * B); upper_idx = int((1 - alpha/2) * B) - 1   (right-exclusive convention, document inline)
       - ci_lower, ci_upper = sorted_diffs[lower_idx], sorted_diffs[upper_idx]
       - point_estimate = statistics.median(x) - statistics.median(y)
       - Return {"ci_lower": ci_lower, "ci_upper": ci_upper, "point_estimate": point_estimate, "B": B, "seed": seed, "degenerate": False, "alpha": alpha}
