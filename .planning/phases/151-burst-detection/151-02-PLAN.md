```python
@dataclass(frozen=True, slots=True)
class BurstResult:
    acceleration: float      # Second derivative: ms/cycle^2
    velocity: float          # First derivative: ms/cycle
    is_burst: bool           # True if burst confirmed
    consecutive_accel: int   # Current streak of above-threshold cycles
    warming_up: bool         # True until 3+ samples

class BurstDetector:
    def __init__(self, wan_name: str, accel_threshold_ms: float,
                 confirm_cycles: int, logger: logging.Logger) -> None: ...
    def update(self, load_rtt: float) -> BurstResult: ...
    def reset(self) -> None: ...
    @property
    def accel_threshold(self) -> float: ...
    @accel_threshold.setter
    def accel_threshold(self, value: float) -> None: ...
    @property
    def confirm_cycles(self) -> int: ...
    @confirm_cycles.setter
    def confirm_cycles(self, value: int) -> None: ...
    @property
    def total_bursts(self) -> int: ...
```

   ```python
       bd = health_data["burst_detection"]
       result = bd["result"]
       return {
           "enabled": bd["enabled"],
           "total_bursts": bd["total_bursts"],
           "burst_response_enabled": False,  # Phase 152 will change this to True
           "current_acceleration": round(result.acceleration, 3) if result else None,
           "current_velocity": round(result.velocity, 3) if result else None,
           "is_burst": result.is_burst if result else False,
           "consecutive_accel_cycles": result.consecutive_accel if result else 0,
           "warming_up": result.warming_up if result else True,
       }
   ```
