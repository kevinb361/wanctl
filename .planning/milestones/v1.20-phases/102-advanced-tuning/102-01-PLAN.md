    ```python
    """Advanced tuning strategies -- fusion weight, reflector scoring, baseline bounds.

    ADVT-01: Fusion ICMP/IRTT weight from per-signal reliability scoring
    ADVT-02: Reflector min_score from signal confidence proxy
    ADVT-03: Baseline RTT bounds from p5/p95 baseline history
    """
    from __future__ import annotations
    import logging
    from statistics import mean, quantiles
    from wanctl.tuning.models import SafetyBounds, TuningResult

    logger = logging.getLogger(__name__)
    MIN_SAMPLES = 60

    # ADVT-02 confidence thresholds for convergence detection
    CONFIDENCE_LOW = 0.5    # Below: min_score too strict
    CONFIDENCE_HIGH = 0.9   # Above: min_score too lenient
    CONFIDENCE_STEP = 0.05  # Adjustment step per cycle

    # ADVT-03 baseline margin multipliers
    BASELINE_MIN_MARGIN = 0.9   # p5 * 0.9
    BASELINE_MAX_MARGIN = 1.1   # p95 * 1.1
    ```

    tune_fusion_weight(metrics_data, current_value, bounds, wan_name) -> TuningResult | None:
    1. Extract wanctl_signal_variance_ms2 values from metrics_data
    2. Extract wanctl_irtt_ipdv_ms values from metrics_data
    3. Extract wanctl_irtt_loss_up_pct and wanctl_irtt_loss_down_pct (average as mean_irtt_loss)
    4. If ICMP variance samples < MIN_SAMPLES, return None
    5. If IRTT jitter samples < MIN_SAMPLES, return None (no comparison possible)
    6. Compute ICMP reliability = 1.0 / (1.0 + mean(icmp_variance_values))
    7. Compute mean IRTT loss = mean of (loss_up + loss_down) / 2 per timestamp, clamped to 0-100 range, divided by 100 to get fraction
    8. Compute IRTT reliability = (1.0 - mean_irtt_loss_fraction) / (1.0 + mean(irtt_ipdv_values))
    9. candidate_icmp_weight = icmp_reliability / (icmp_reliability + irtt_reliability)
    10. confidence = min(1.0, min(len(icmp_vars), len(irtt_jitters)) / 1440.0)
    11. Return TuningResult(parameter="fusion_icmp_weight", ...)

    tune_reflector_min_score(metrics_data, current_value, bounds, wan_name) -> TuningResult | None:
    1. Extract wanctl_signal_confidence values
    2. If < MIN_SAMPLES, return None
    3. Compute mean_confidence = mean(confidence_values)
    4. If CONFIDENCE_LOW <= mean_confidence <= CONFIDENCE_HIGH, return None (converged)
    5. If mean_confidence < CONFIDENCE_LOW: candidate = current_value - CONFIDENCE_STEP (loosen)
    6. If mean_confidence > CONFIDENCE_HIGH: candidate = current_value + CONFIDENCE_STEP (tighten)
    7. confidence = min(1.0, len(confidence_values) / 1440.0)
    8. Return TuningResult(parameter="reflector_min_score", ...)

    tune_baseline_bounds_min(metrics_data, current_value, bounds, wan_name) -> TuningResult | None:
    1. Extract wanctl_rtt_baseline_ms values
    2. If < MIN_SAMPLES, return None
    3. Compute percentiles = quantiles(baseline_values, n=100)
    4. p5 = percentiles[4]  # 5th percentile
    5. candidate = p5 * BASELINE_MIN_MARGIN
    6. candidate = max(candidate, 1.0)  # Hard floor
    7. confidence = min(1.0, len(baseline_values) / 1440.0)
    8. Return TuningResult(parameter="baseline_rtt_min", ...)

    tune_baseline_bounds_max(metrics_data, current_value, bounds, wan_name) -> TuningResult | None:
    1. Extract wanctl_rtt_baseline_ms values
    2. If < MIN_SAMPLES, return None
    3. Compute percentiles = quantiles(baseline_values, n=100)
    4. p95 = percentiles[94]  # 95th percentile
    5. candidate = p95 * BASELINE_MAX_MARGIN
    6. confidence = min(1.0, len(baseline_values) / 1440.0)
    7. Return TuningResult(parameter="baseline_rtt_max", ...)
