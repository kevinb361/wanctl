       import argparse, httpx
       from textual.app import App, ComposeResult
       from textual.binding import Binding
       from textual.containers import Vertical
       from wanctl.dashboard.config import DashboardConfig, load_dashboard_config, apply_cli_overrides
       from wanctl.dashboard.poller import EndpointPoller
       from wanctl.dashboard.widgets import WanPanel, SteeringPanel, StatusBar

       class DashboardApp(App):
           CSS_PATH = "dashboard.tcss"

           BINDINGS = [
               Binding("q", "quit", "Quit"),
               Binding("r", "refresh", "Refresh"),
           ]

           def __init__(self, config: DashboardConfig):
               super().__init__()
               self.config = config
               self._autorate_poller = EndpointPoller("autorate", config.autorate_url, normal_interval=config.refresh_interval)
               self._steering_poller = EndpointPoller("steering", config.steering_url, normal_interval=config.refresh_interval)
               self._client: httpx.AsyncClient | None = None

           def compose(self) -> ComposeResult:
               with Vertical():
                   yield WanPanel("WAN 1 (Spectrum)", id="wan-1")
                   yield WanPanel("WAN 2 (ATT)", id="wan-2")
                   yield SteeringPanel(id="steering")
                   yield StatusBar(id="status-bar")

           async def on_mount(self):
               self._client = httpx.AsyncClient(timeout=2.0)
               self.set_interval(self.config.refresh_interval, self._poll_autorate)
               self.set_interval(self.config.refresh_interval, self._poll_steering)

           async def on_unmount(self):
               if self._client:
                   await self._client.aclose()

           async def _poll_autorate(self):
               data = await self._autorate_poller.poll(self._client)
               wan1 = self.query_one("#wan-1", WanPanel)
               wan2 = self.query_one("#wan-2", WanPanel)
               if data and "wans" in data:
                   wans = data["wans"]
                   if len(wans) >= 1:
                       wan1.update_from_data(wans[0], status=data.get("status"), last_seen=self._autorate_poller.last_seen)
                   if len(wans) >= 2:
                       wan2.update_from_data(wans[1], status=data.get("status"), last_seen=self._autorate_poller.last_seen)
                   status_bar = self.query_one("#status-bar", StatusBar)
                   status_bar.update(
                       version=data.get("version", "?"),
                       uptime_seconds=data.get("uptime_seconds", 0),
                       disk_status=data.get("disk_space", {}).get("status", "unknown"),
                   )
               else:
                   wan1.update_from_data(None, last_seen=self._autorate_poller.last_seen)
                   wan2.update_from_data(None, last_seen=self._autorate_poller.last_seen)

           async def _poll_steering(self):
               data = await self._steering_poller.poll(self._client)
               panel = self.query_one("#steering", SteeringPanel)
               if data:
                   panel.update_from_data(data, last_seen=self._steering_poller.last_seen)
               else:
                   panel.update_from_data(None, last_seen=self._steering_poller.last_seen)

           async def action_refresh(self):
               await self._poll_autorate()
               await self._poll_steering()

       def parse_args(argv: list[str] | None = None) -> argparse.Namespace:
           parser = argparse.ArgumentParser(description="wanctl TUI Dashboard")
           parser.add_argument("--autorate-url", default=None)
           parser.add_argument("--steering-url", default=None)
           parser.add_argument("--refresh-interval", type=float, default=None)
           parser.add_argument("--config", type=str, default=None)
           return parser.parse_args(argv)

       def main(argv: list[str] | None = None):
           args = parse_args(argv)
           config = load_dashboard_config(Path(args.config) if args.config else None)
           config = apply_cli_overrides(config, args)
           app = DashboardApp(config)
           app.run()
