    Create the module with these exact types and functions:

    ```python
    U32_MAX = 0xFFFFFFFF
    CYCLE_INTERVAL_SECONDS = 0.05
    SANITY_MAX_DELTA = 1_000_000  # >1M drops in 50ms is impossible

    def u32_delta(current: int, previous: int) -> int:
        """Compute delta between two u32 counters, handling wrap-around.
        Returns 0 if delta exceeds SANITY_MAX_DELTA (wrap artifact)."""

    @dataclass(frozen=True, slots=True)
    class TinSnapshot:
        name: str
        dropped_packets: int   # Raw counter (cumulative)
        drop_delta: int        # Delta since last read
        backlog_bytes: int
        peak_delay_us: int
        ecn_marked_packets: int

    @dataclass(frozen=True, slots=True)
    class CakeSignalSnapshot:
        drop_rate: float           # EWMA drops/sec (BestEffort+ tins only, excludes Bulk)
        total_drop_rate: float     # EWMA drops/sec (all tins including Bulk)
        backlog_bytes: int         # Sum of BestEffort+ tins backlog
        peak_delay_us: int         # Max peak_delay across BestEffort+ tins
        tins: tuple[TinSnapshot, ...]
        cold_start: bool

    @dataclass(frozen=True, slots=True)
    class CakeSignalConfig:
        enabled: bool = False
        drop_rate_enabled: bool = False
        backlog_enabled: bool = False
        peak_delay_enabled: bool = False
        metrics_enabled: bool = False
        time_constant_sec: float = 1.0

    class CakeSignalProcessor:
        def __init__(self, config: CakeSignalConfig, tin_names: list[str] | None = None):
            self._config = config
            self._tin_names = tin_names or ["Bulk", "BestEffort", "Video", "Voice"]
            self._prev_counters: dict[int, int] | None = None  # tin_index -> dropped_packets
            self._cold_start = True
            self._drop_rate_ewma = 0.0       # active (excl Bulk)
            self._total_drop_rate_ewma = 0.0  # all tins
            self._alpha = CYCLE_INTERVAL_SECONDS / config.time_constant_sec
            self._last_snapshot: CakeSignalSnapshot | None = None

        @property
        def config(self) -> CakeSignalConfig: ...
        @config.setter
        def config(self, value: CakeSignalConfig) -> None:
            """Update config (for SIGUSR1 reload). Recalculates alpha."""

        def update(self, raw_stats: dict[str, Any] | None) -> CakeSignalSnapshot | None:
            """Process one cycle of CAKE stats. Returns None if disabled or stats unavailable."""
    ```

    Implementation details for update():
    1. If not self._config.enabled, return None
    2. If raw_stats is None, return self._last_snapshot (no crash)
    3. Extract per-tin dropped_packets from raw_stats["tins"]
    4. If self._prev_counters is None (cold start): store counters, return snapshot with cold_start=True, zero rates
    5. Compute u32_delta for each tin
    6. Sum active_drops (tins[1:] = BestEffort+Video+Voice), total_drops (all tins)
    7. Convert to drops/sec: active_drops / CYCLE_INTERVAL_SECONDS
    8. EWMA smooth: self._drop_rate_ewma = (1 - alpha) * self._drop_rate_ewma + alpha * active_drops_per_sec
    9. Same for total_drop_rate_ewma
    10. Build TinSnapshot tuple from raw stats + deltas
    11. Compute backlog_bytes = sum of tins[1:] backlog_bytes
    12. Compute peak_delay_us = max of tins[1:] peak_delay_us
    13. Build and store CakeSignalSnapshot, return it

    ```python
    def _run_cake_stats(self) -> None:
        """Read CAKE qdisc stats and update signal processors (Phase 159).

        Runs on main thread (IPRoute not thread-safe). Reads stats from
        NetlinkCakeBackend.get_queue_stats() for both DL and UL directions.
        Only active when transport is linux-cake and cake_signal is enabled.

        from wanctl.backends.linux_cake_adapter import LinuxCakeAdapter

        dl_stats = adapter.dl_backend.get_queue_stats("")
        self._dl_cake_snapshot = self._dl_cake_signal.update(dl_stats)

        ul_stats = adapter.ul_backend.get_queue_stats("")
        self._ul_cake_snapshot = self._ul_cake_signal.update(ul_stats)
    ```
