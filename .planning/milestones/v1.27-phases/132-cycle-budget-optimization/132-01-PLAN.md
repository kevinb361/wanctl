<objective>
Decouple RTT measurement from the control loop by moving ICMP pings to a dedicated
background thread with a persistent ThreadPoolExecutor. The control loop reads the
latest RTT from a GIL-protected frozen dataclass pointer swap instead of blocking
on 42ms of ICMP I/O every cycle.

Purpose: Eliminate the 84.6% cycle budget bottleneck (42ms avg under RRUL load) so
the control loop drops from ~51ms to ~10ms, achieving <80% utilization on a 50ms
cycle interval.

Output: BackgroundRTTThread class in rtt_measurement.py, non-blocking measure_rtt()
in autorate_continuous.py, full test coverage, clean daemon lifecycle integration.
</objective>

```python
class IRTTThread:
    def __init__(self, measurement, cadence_sec, shutdown_event, logger):
        self._cached_result: IRTTResult | None = None  # GIL-protected atomic swap
        self._thread: threading.Thread | None = None

    def get_latest(self) -> IRTTResult | None:
        return self._cached_result  # Lock-free read

    def start(self) -> None:
        self._thread = threading.Thread(target=self._run, name="wanctl-irtt", daemon=True)
        self._thread.start()

    def stop(self) -> None:
        if self._thread is not None:
            self._thread.join(timeout=5.0)

    def _run(self) -> None:
        while not self._shutdown_event.is_set():
            result = self._measurement.measure()
            if result is not None:
                self._cached_result = result
            self._shutdown_event.wait(timeout=self._cadence_sec)
```
