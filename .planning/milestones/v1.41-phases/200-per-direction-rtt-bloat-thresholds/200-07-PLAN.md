    Step 2 — Author `scripts/phase200-soak-capture.sh`. Sub-modes:
    - `start` — initializes run dir, launches background capture, returns run-dir path
    - `status <RUN_DIR>` — reports elapsed time and current sample count
    - `finalize <RUN_DIR>` — touches capture-completed.flag, hands off to analyze

    ```bash
    #!/usr/bin/env bash
    # Phase 200 24h regression soak capture (VALN-06 watchdog).
    # Modes: start | status <RUN_DIR> | finalize <RUN_DIR>
    set -euo pipefail

    print_help() {
        cat <<'HELP'
    Phase 200 24h regression soak.

    Usage:
      scripts/phase200-soak-capture.sh start
        Initializes ${PHASE200_OUT_DIR}/soak/<UTC-TS>/, launches background
        capture (1Hz, hourly-rotated NDJSON), prints RUN_DIR.

      scripts/phase200-soak-capture.sh status <RUN_DIR>
        Reports elapsed time and current sample counts.

      scripts/phase200-soak-capture.sh finalize <RUN_DIR>
        Touches capture-completed.flag. Run once duration has elapsed.

    Required env vars (start mode):
      PHASE200_OUT_DIR
      PHASE200_SPECTRUM_HEALTH_URL

    Optional overrides:
      PHASE200_SOAK_DURATION_HOURS    Default 24
      PHASE200_SOAK_POLL_HZ           Default 1
    HELP
    }

    [[ "${1:-}" == "--help" || "${1:-}" == "-h" ]] && { print_help; exit 0; }

    MODE="${1:-}"
    [[ -n "$MODE" ]] || { print_help; exit 2; }

    case "$MODE" in
      start)
        : "${PHASE200_OUT_DIR:?env var required}"
        : "${PHASE200_SPECTRUM_HEALTH_URL:?env var required}"
        DURATION_H="${PHASE200_SOAK_DURATION_HOURS:-24}"
        POLL_HZ="${PHASE200_SOAK_POLL_HZ:-1}"
        UTC_TS="$(date -u +%Y%m%dT%H%M%SZ)"
        RUN_DIR="${PHASE200_OUT_DIR}/soak/${UTC_TS}"
        mkdir -p "$RUN_DIR"
        echo "$DURATION_H" > "${RUN_DIR}/duration_hours.txt"
        echo "$PHASE200_SPECTRUM_HEALTH_URL" > "${RUN_DIR}/health_url.txt"
        # Pin the actual sampling rate so the analyzer can convert sample-counts
        # to seconds correctly even if the operator overrides PHASE200_SOAK_POLL_HZ
        # for a future re-run. Without this file the analyzer would silently assume 1Hz.
        echo "$POLL_HZ" > "${RUN_DIR}/poll_hz.txt"
        # Background capture loop — writes one NDJSON file per hour-of-soak.
        # Use nohup + & rather than systemd-run to avoid OS-coupling.
        nohup bash -c '
            set -euo pipefail
            URL="'"$PHASE200_SPECTRUM_HEALTH_URL"'"
            DUR_H='"$DURATION_H"'
            POLL_HZ='"$POLL_HZ"'
            INTERVAL=$(awk "BEGIN{print 1/${POLL_HZ}}")
            END=$(( $(date +%s) + DUR_H * 3600 ))
            START_HOUR=$(date +%s)
            HOUR_NUM=0
            while [[ $(date +%s) -lt $END ]]; do
                HOUR_FILE="'"$RUN_DIR"'/capture-hour-$(printf %02d $HOUR_NUM).ndjson"
                HOUR_END=$(( $(date +%s) + 3600 ))
                if [[ $HOUR_END -gt $END ]]; then HOUR_END=$END; fi
                while [[ $(date +%s) -lt $HOUR_END ]]; do
                    curl --silent --max-time 2 "$URL" \
                        | jq -c --arg ts "$(date -u +%Y-%m-%dT%H:%M:%SZ)" ". + {sampled_at_utc: \$ts}" \
                        >> "$HOUR_FILE" || true
                    sleep "$INTERVAL"
                done
                HOUR_NUM=$((HOUR_NUM + 1))
            done
            touch "'"$RUN_DIR"'/capture-completed.flag"
        ' > "${RUN_DIR}/nohup.log" 2>&1 &
        BG_PID=$!
        echo "$BG_PID" > "${RUN_DIR}/capture.pid"
        echo "Started Phase 200 soak capture pid=${BG_PID} run_dir=${RUN_DIR}"
        echo "$RUN_DIR"
        ;;
      status)
        RUN_DIR="${2:?usage: status <RUN_DIR>}"
        [[ -d "$RUN_DIR" ]] || { echo "ABORT: $RUN_DIR not a directory" >&2; exit 2; }
        echo "Run dir: $RUN_DIR"
        if [[ -f "${RUN_DIR}/capture.pid" ]]; then
            PID="$(cat "${RUN_DIR}/capture.pid")"
            if kill -0 "$PID" 2>/dev/null; then
                echo "Capture running (pid=$PID)"
            else
                echo "Capture process not running"
            fi
        fi
        echo "Hourly files:"
        ls -la "${RUN_DIR}"/capture-hour-*.ndjson 2>/dev/null || echo "  (none yet)"
        echo "Completed flag: $(test -f "${RUN_DIR}/capture-completed.flag" && echo present || echo absent)"
        ;;
      finalize)
        RUN_DIR="${2:?usage: finalize <RUN_DIR>}"
        touch "${RUN_DIR}/capture-completed.flag"
        echo "Finalized: ${RUN_DIR}/capture-completed.flag"
        ;;
      *)
        echo "Unknown mode: $MODE" >&2
        print_help
        exit 2
        ;;
    esac
    ```

    Step 3 — Author `scripts/phase200-soak-analyze.py`. Reads all `capture-hour-*.ndjson` under a run dir, computes:
    - mean and p95 of `wans[0].upload.hysteresis.suppressions_last_60s` across the full window
    - total seconds the upload was at floor (`current_rate_mbps == floor_mbps`)
    - state distribution (% time in GREEN / YELLOW / RED)
    - PASS if mean < `PHASE200_SOAK_HYSTERESIS_PASS_THRESHOLD` (default 5/60s)

    ```python
    #!/usr/bin/env python3
    """Phase 200 24h regression soak analyzer (VALN-06 watchdog).

    Reads capture-hour-*.ndjson under a soak run dir, emits soak-verdict.json
    with mean/p95 UL hysteresis suppression rate, UL floor seconds, state
    distribution, and pass/fail verdict.
    """
    from __future__ import annotations

    import argparse
    import json
    import statistics
    import sys
    from pathlib import Path


    def parse_args() -> argparse.Namespace:
        p = argparse.ArgumentParser(description=__doc__)
        p.add_argument("run_dir", type=Path, help="Soak run directory")
        p.add_argument(
            "--pass-threshold-per-60s",
            type=float,
            default=5.0,
            help="Mean UL hysteresis suppressions_last_60s strictly below this -> PASS (default 5)",
        )
        return p.parse_args()


    def iter_samples(run_dir: Path):
        for ndjson in sorted(run_dir.glob("capture-hour-*.ndjson")):
            with ndjson.open() as f:
                for line in f:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        yield json.loads(line)
                    except json.JSONDecodeError:
                        continue


    def main() -> int:
        args = parse_args()
        run_dir: Path = args.run_dir
        if not run_dir.is_dir():
            print(f"ABORT: {run_dir} is not a directory", file=sys.stderr)
            return 2

        # Read pinned poll rate from the run dir (capture script writes this at
        # `start` time). Falls back to 1Hz for backward compat with any pre-Plan-07
        # captures, but emits a WARNING if absent so the operator knows seconds
        # arithmetic is approximate.
        poll_hz_path = run_dir / "poll_hz.txt"
        if poll_hz_path.is_file():
            try:
                poll_hz = float(poll_hz_path.read_text().strip())
            except ValueError:
                print(f"WARN: {poll_hz_path} unparseable; defaulting to 1Hz", file=sys.stderr)
                poll_hz = 1.0
        else:
            print(f"WARN: {poll_hz_path} not found; defaulting to 1Hz (seconds may be approximate)", file=sys.stderr)
            poll_hz = 1.0
        if poll_hz <= 0:
            print(f"WARN: invalid poll_hz={poll_hz}; defaulting to 1Hz", file=sys.stderr)
            poll_hz = 1.0

        suppressions: list[int] = []
        floor_samples = 0
        state_counts: dict[str, int] = {"GREEN": 0, "YELLOW": 0, "RED": 0, "OTHER": 0}
        total_samples = 0
        first_ts: str | None = None
        last_ts: str | None = None

        for sample in iter_samples(run_dir):
            try:
                wan0 = sample["wans"][0]
                up = wan0["upload"]
                supp = up.get("hysteresis", {}).get("suppressions_last_60s")
                if isinstance(supp, (int, float)):
                    suppressions.append(int(supp))
                if up.get("current_rate_mbps") is not None and up.get("floor_mbps") is not None:
                    if up["current_rate_mbps"] == up["floor_mbps"]:
                        floor_samples += 1
                state = up.get("state", "OTHER")
                state_counts[state if state in state_counts else "OTHER"] += 1
                total_samples += 1
                ts = sample.get("sampled_at_utc")
                if ts:
                    if first_ts is None:
                        first_ts = ts
                    last_ts = ts
            except (KeyError, IndexError, TypeError):
                continue

        if total_samples == 0:
            verdict = {
                "phase": 200,
                "run_dir": str(run_dir),
                "verdict": "abort",
                "reason": "no_valid_samples",
            }
        else:
            mean_supp = statistics.mean(suppressions) if suppressions else 0.0
            try:
                p95_supp = statistics.quantiles(suppressions, n=20)[18] if len(suppressions) > 20 else max(suppressions)
            except statistics.StatisticsError:
                p95_supp = 0.0
            verdict_val = "pass" if mean_supp < args.pass_threshold_per_60s else "fail"
            verdict = {
                "phase": 200,
                "run_dir": str(run_dir),
                "started_at_utc": first_ts,
                "finished_at_utc": last_ts,
                "duration_sec": total_samples / poll_hz,
                "poll_hz": poll_hz,
                "ul_hysteresis_per_60s_mean": mean_supp,
                "ul_hysteresis_per_60s_p95": p95_supp,
                "ul_floor_seconds_total": floor_samples / poll_hz,
                "ul_state_distribution": {
                    k: round(v / total_samples * 100, 2) for k, v in state_counts.items()
                },
                "pass_threshold_per_60s": args.pass_threshold_per_60s,
                "verdict": verdict_val,
                "rollback_protocol": (
                    "Per D-10: revert /opt/wanctl to v1.40 binary on cake-shaper. "
                    "tarball at /opt/wanctl-prephase200-<UTC_TS>.tar.gz. "
                    "/etc/wanctl/spectrum.yaml stays in place — pre-Plan-01 binary silently no-ops the new keys."
                ),
            }

        out_path = run_dir / "soak-verdict.json"
        out_path.write_text(json.dumps(verdict, indent=2))
        print(json.dumps(verdict, indent=2))
        return 0 if verdict.get("verdict") == "pass" else 1


    if __name__ == "__main__":
        sys.exit(main())
    ```
