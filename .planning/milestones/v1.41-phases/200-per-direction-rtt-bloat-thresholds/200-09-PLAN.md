```bash
set -o pipefail
NDJSON=.planning/phases/200-per-direction-rtt-bloat-thresholds/canary/20260503T215734Z/loaded_capture.ndjson

# A) Floor-sample run lengths (consecutive seconds where current_rate_mbps == 8.0)
# Round-2 HIGH-1 fix: jq 1.7 rejects `// null` inside object literals without parens.
# Drop `// null` entirely — every field is verified-present at top level / wans[0] / wans[0].upload.
jq -c '{ts: .sampled_at_utc, rate: .wans[0].upload.current_rate_mbps, state: .wans[0].upload.state, base: .wans[0].baseline_rtt_ms, load: .wans[0].load_rtt_ms}' "$NDJSON" \
  | python3 -c '
import sys, json
runs = []
cur = 0
for line in sys.stdin:
    d = json.loads(line)
    if d["rate"] == 8.0:
        cur += 1
    else:
        if cur > 0: runs.append(cur)
        cur = 0
if cur > 0: runs.append(cur)
print("floor_run_lengths:", runs)
print("count:", len(runs), "max:", max(runs) if runs else 0, "total_floor_samples:", sum(runs))
'

# B) State-transition adjacency matrix (GREEN / YELLOW / RED counts pair-by-pair)
jq -r '.wans[0].upload.state' "$NDJSON" \
  | python3 -c '
import sys
from collections import Counter
states = [s.strip() for s in sys.stdin if s.strip()]
trans = Counter(zip(states, states[1:]))
print("transitions:", dict(trans))
print("counts:", dict(Counter(states)))
'

# C) RTT-delta percentiles bucketed by zone
jq -c '{ts: .sampled_at_utc, baseline: .wans[0].baseline_rtt_ms, load: .wans[0].load_rtt_ms, state: .wans[0].upload.state}' "$NDJSON" \
  | python3 -c '
import sys, json, statistics
from collections import defaultdict
buckets = defaultdict(list)
for line in sys.stdin:
    d = json.loads(line)
    if d["baseline"] and d["load"]:
        buckets[d["state"]].append(d["load"] - d["baseline"])
for state, vals in buckets.items():
    if vals:
        q = statistics.quantiles(vals, n=20)
        print(f"{state}: n={len(vals)} p50={q[9]:.1f} p90={q[17]:.1f} p95={q[18]:.1f} max={max(vals):.1f}")
'

# D) Distribution of YELLOW samples vs floor samples (does floor coincide with YELLOW or RED?)
jq -c '{rate: .wans[0].upload.current_rate_mbps, state: .wans[0].upload.state}' "$NDJSON" \
  | python3 -c '
import sys, json
from collections import Counter
c = Counter()
for line in sys.stdin:
    d = json.loads(line)
    floor = d["rate"] == 8.0
    c[(d["state"], "FLOOR" if floor else "above_floor")] += 1
print(dict(c))
'

# E) Time-to-first-floor from start of loaded window (seconds)
jq -c '{ts: .sampled_at_utc, rate: .wans[0].upload.current_rate_mbps}' "$NDJSON" \
  | python3 -c '
import sys, json
first_ts = None
first_floor = None
for line in sys.stdin:
    d = json.loads(line)
    if first_ts is None:
        first_ts = d["ts"]
    if d["rate"] == 8.0 and first_floor is None:
        first_floor = d["ts"]
        break
print("first_load_sample:", first_ts, "first_floor_sample:", first_floor)
'

# F) Cross-reference: are floor samples in YELLOW state, RED state, or after a RED that decayed?
# Goal: determine whether floor samples are predominantly preceded by ≥3 consecutive YELLOW
# samples (R5 ship-decision evidence checklist item 1) or arise from RED transitions.
jq -c '{rate: .wans[0].upload.current_rate_mbps, state: .wans[0].upload.state}' "$NDJSON" \
  | python3 -c '
import sys, json
from collections import Counter
samples = [json.loads(line) for line in sys.stdin]
preceded_by_yellow_3 = 0
preceded_by_red = 0
floor_total = 0
for i, s in enumerate(samples):
    if s["rate"] == 8.0:
        floor_total += 1
        prev3 = samples[max(0, i-3):i]
        if prev3 and all(p["state"] == "YELLOW" for p in prev3):
            preceded_by_yellow_3 += 1
        if prev3 and any(p["state"] == "RED" for p in prev3):
            preceded_by_red += 1
print(f"floor_total={floor_total} preceded_by_yellow_3={preceded_by_yellow_3} preceded_by_red={preceded_by_red}")
'
# Sample expected output: floor_total=122 preceded_by_yellow_3=98 preceded_by_red=4

# G) True-YELLOW (deadband) vs below-target vs above-warn breakdown.
# Round-2 MEDIUM fix: this WAS a comment, now it's an executable command with thresholds wired in.
# Thresholds 42 / 105 are Spectrum's target_bloat_ms / warn_bloat_ms (configs/spectrum.yaml:74-75).
# Differentiates: true-deadband YELLOW (42 < delta <= 105) vs below-target cling (delta < 42)
# vs above-warn anomaly (delta > 105 but state still YELLOW).
jq -c 'select(.wans[0].upload.state == "YELLOW") | {delta: (.wans[0].load_rtt_ms - .wans[0].baseline_rtt_ms), state_reason: .wans[0].upload.state_reason}' "$NDJSON" \
  | awk -F'"delta":' '{ split($2, a, ","); d = a[1]+0; if (d < 42) below_target++; else if (d <= 105) deadband++; else above_warn++ } END { printf "true-yellow-deadband (target<delta<=warn): %d, below-target: %d, above-warn: %d\n", deadband, below_target, above_warn }'
# Sample expected output: true-yellow-deadband (target<delta<=warn): 412, below-target: 89, above-warn: 23
```
